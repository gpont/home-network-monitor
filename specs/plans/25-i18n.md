## Реализация

```typescript
// index.ts
import { writable, derived } from 'svelte/store';
import ru from './ru.ts';
import en from './en.ts';

export type Locale = 'ru' | 'en';

export function detectLocale(): Locale {
  const saved = localStorage.getItem('locale') as Locale;
  if (saved === 'ru' || saved === 'en') return saved;
  return navigator.language.toLowerCase().startsWith('ru') ? 'ru' : 'en';
}

export const locale = writable<Locale>(detectLocale());
locale.subscribe(l => localStorage.setItem('locale', l));

const dicts: Record<Locale, Record<string, string>> = { ru, en };

export const t = derived(locale, $locale =>
  (key: string, vars?: Record<string, string | number>): string => {
    let text = dicts[$locale][key] ?? dicts['ru'][key] ?? key;
    if (vars) {
      for (const [k, v] of Object.entries(vars)) {
        text = text.replace(`{${k}}`, String(v));
      }
    }
    return text;
  }
);
```

## Тесты (9 штук)

- `t()` возвращает RU-текст при `locale = 'ru'`
- `t()` возвращает EN-текст при `locale = 'en'`
- `t()` fallback на RU если ключ отсутствует в EN
- `t()` fallback на raw key если нет ни в EN, ни в RU
- `t()` интерполирует `{n}` переменные
- `detectLocale()` читает значение из `localStorage`
- `detectLocale()` возвращает `'ru'` для `navigator.language = 'ru-RU'`
- `detectLocale()` возвращает `'en'` для `navigator.language = 'en-US'`
- `detectLocale()` возвращает `'en'` если localStorage пуст и язык не русский

> **Примечание:** `localStorage` мокировать до импорта модуля.
